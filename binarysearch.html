Recursive Binary Search Visualization

Algorithm:

int binarySearch(int arr[], int left, int right, int key) {

    if (left > right)
        return -1;

    int mid = (left + right) / 2;

    if (arr[mid] == key)
        return mid;

    else if (key < arr[mid])
        return binarySearch(arr, left, mid - 1, key);

    else
        return binarySearch(arr, mid + 1, right, key);
}
